Basic Setup

// Import core styles import './ui-components/core/tokens.css'; import './ui-components/core/utilities.css'; // Import components import './ui-components/atoms/tmyl-button.js'; import './ui-components/molecules/tmyl-search-bar.js'; // Use in HTML <tmyl-search-bar voice-enabled></tmyl-search-bar>

Event Handling

// Listen to component events searchInterface.addEventListener('tmyl-search', (e) => { console.log('Search query:', e.detail.query); }); searchInterface.addEventListener('tmyl-content-select', (e) => { console.log('Selected:', e.detail.content); });
